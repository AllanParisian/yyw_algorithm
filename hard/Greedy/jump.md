public class jump {
    public int jump(int[] nums) {
        int n = nums.length;
        int[] count = new int[n];
        int start = 0;
        for(int i = 0; i < n && start < n; i++){
            if(i + nums[i] < start)
                continue;
            for(int j = start; j < n && j <= i + nums[i]; j++)
                count[j] = count[i] + 1;
            start = i + nums[i] + 1;
        }
        return count[n - 1] - 1;
    }
}

            public int jump(int[] nums) {
                int n = nums.length;
                int[] count = new int[n];
                Arrays.fill(count, Integer.MAX_VALUE);
                count[0] = 0;
                for(int i = 1; i < n; i++){

                    }
